import { Injectable } from "@nestjs/common";
import { Context, Markup } from "telegraf";
import { InjectModel } from "@nestjs/mongoose";
import { Model } from "mongoose";
import { Bot, BotDocument } from "./schema/bot.schema";

@Injectable()
export class BotService {
  private kasallikMap = new Map<number, boolean>();

  constructor(
    @InjectModel(Bot.name) private readonly botModel: Model<BotDocument>
  ) {}

  // /start komandasi
  async start(ctx: Context) {
    await ctx.reply("Assalomu alaykum! Telefon raqamingizni yuboring:", {
      reply_markup: {
        keyboard: [
          [{ text: "ğŸ“ Telefon raqamni yuborish", request_contact: true }],
        ],
        resize_keyboard: true,
        one_time_keyboard: true,
      },
    });
  }

  // Telefon raqamini qabul qilish
  async onContact(ctx: Context) {
    const user_id = ctx.from?.id;
    if (!user_id || !ctx.message) {
      return ctx.reply("â—ï¸ Foydalanuvchi yoki xabar aniqlanmadi.");
    }

    if ("contact" in ctx.message && ctx.message.contact?.phone_number) {
      const contact = ctx.message.contact;
      let user = await this.botModel.findOne({ user_id });

      if (user) {
        user.phone_number = contact.phone_number;
        await user.save();
      } else {
        await this.botModel.create({
          user_id,
          phone_number: contact.phone_number,
          user_name: ctx.from?.username ?? "",
          first_name: ctx.from?.first_name ?? "",
          last_name: ctx.from?.last_name ?? "",
          lang: ctx.from?.language_code ?? "",
        });
      }

      await ctx.reply("âœ… Endi ismingizni yuboring:", {
        reply_markup: { remove_keyboard: true },
      });
    } else {
      await ctx.reply("â—ï¸ Telefon raqami aniqlanmadi.");
    }
  }

  // Ismni qabul qilish
  async onName(ctx: Context) {
    const user_id = ctx.from?.id;
    if (!user_id || !ctx.message) return;

    if ("text" in ctx.message && typeof ctx.message.text === "string") {
      const text = ctx.message.text.trim();

      if (!/^[a-zA-Z\u0400-\u04FF\s'-]{2,}$/.test(text)) {
        return ctx.reply(
          "â—ï¸ Iltimos, faqat harflardan iborat ismingizni yozing."
        );
      }

      const existingUser = await this.botModel.findOne({ user_id });

      if (existingUser) {
        if (!existingUser.name) {
          existingUser.name = text;
          await existingUser.save();
          await ctx.reply(`âœ… Ismingiz saqlandi: ${text}`);
        }
      } else {
        await this.botModel.create({
          user_id,
          name: text,
          user_name: ctx.from?.username ?? "",
          first_name: ctx.from?.first_name ?? "",
          last_name: ctx.from?.last_name ?? "",
          lang: ctx.from?.language_code ?? "",
        });

        await ctx.reply(`âœ… Ismingiz saqlandi: ${text}`);
      }

      // Lokatsiya soâ€˜rash
      await ctx.reply("ğŸ“ Endi joylashuvingizni yuboring:", {
        reply_markup: {
          keyboard: [
            [{ text: "ğŸ“ Joylashuvni yuborish", request_location: true }],
          ],
          resize_keyboard: true,
          one_time_keyboard: true,
        },
      });
    } else {
      await ctx.reply("âŒ Iltimos, faqat matn yuboring.");
    }
  }

  // Joylashuvni qabul qilish
  async onLocation(ctx: Context) {
    const user_id = ctx.from?.id;
    if (!user_id || !ctx.message) {
      return ctx.reply("â—ï¸ Foydalanuvchi yoki xabar aniqlanmadi.");
    }

    // Faqat location kelgandagina saqlaymiz
    if ("location" in ctx.message && ctx.message.location) {
      const { latitude, longitude } = ctx.message.location;
      const user = await this.botModel.findOne({ user_id });

      if (!user) {
        return ctx.reply("Iltimos, avval /start buyrug'ini bosing.");
      }

      const location = `${latitude},${longitude}`;
      user.location = location;
      await user.save();

      await ctx.reply("âœ… Joylashuvingiz saqlandi.");
      return this.kasallikniSozlash(ctx);
    }

    // âš ï¸ Aks holda notoâ€˜gâ€˜ri xabar yuborilgan boâ€˜ladi (matn, sticker, rasm...)
    // return ctx.reply("â—ï¸ Iltimos, tugmani bosib joylashuv yuboring!", {
    //   reply_markup: {
    //     keyboard: [
    //       [{ text: "ğŸ“ Joylashuvni yuborish", request_location: true }],
    //     ],
    //     resize_keyboard: true,
    //     one_time_keyboard: true,
    //   },
    // });
    return ctx.reply("Yo lokatsiya ");
  }

  // Kasallik haqida tugmalar
  async kasallikniSozlash(ctx: Context) {
    await ctx.reply("ğŸ©º Sizda hozirda biron kasallik bormi?", {
      reply_markup: {
        keyboard: [["âœ… Ha", "âŒ Yoâ€˜q"]],
        resize_keyboard: true,
        one_time_keyboard: true,
      },
    });
  }

  // Matnli javoblar uchun umumiy handler
  async handleText(ctx: Context) {
    const user_id = ctx.from?.id;
    if (!ctx.message || !("text" in ctx.message)) return;
    const text = ctx.message.text.trim();
    if (!user_id || !text) return;

    const user = await this.botModel.findOne({ user_id });

    // Ism hali yoâ€˜q boâ€˜lsa, ismni saqlash
    if (user && !user.name) {
      return this.onName(ctx);
    }

    // âŒ Yoâ€˜q -> sog'lom deb saqlash
    if (text === "âŒ Yoâ€˜q") {
      if (user) {
        user.illness = "sog'lom";
        await user.save();
      }
      await ctx.reply("âœ… Siz sog'lom deb belgilandingiz.", {
        reply_markup: Markup.removeKeyboard().reply_markup,
      });
      return this.nextStep(ctx);
    }

    // âœ… Ha -> kasallik soâ€˜rash
    if (text === "âœ… Ha") {
      this.kasallikMap.set(user_id, true); // â¤ Bu foydalanuvchidan kasallik haqida matn kutamiz degani
      return ctx.reply(
        "ğŸ“ Qanday kasallik borligini yozing (masalan: gripp, allergiya)",
        { reply_markup: Markup.removeKeyboard().reply_markup } // â¤ Keyboard'ni tozalab yuborish
      );
    }

    // Foydalanuvchi kasallik nomini yozgan boâ€˜lsa
    if (this.kasallikMap.has(user_id)) {
      if (user) {
        user.illness = text; // qanday yozgan boâ€˜lsa, shunday saqlaymiz
        await user.save();
        this.kasallikMap.delete(user_id);
        await ctx.reply(`âœ… Kasalligingiz saqlandi: ${text}`);
        return this.nextStep(ctx);
      }
    }
    return ctx.reply("â“ Noma'lum buyruq. Iltimos, tugmani bosing !");
  }

  // Keyingi bosqich (kelajakda xizmat turi, manzil, va h.k.)
  async nextStep(ctx: Context) {
    await ctx.reply(
      "ğŸš€ Roâ€˜yxatdan oâ€˜tish tugadi! Yaqinda xizmatni boshlaysiz."
    );
  }
}

import { Injectable } from "@nestjs/common";
import { Context, Markup } from "telegraf";
import { InjectModel } from "@nestjs/mongoose";
import { Model } from "mongoose";
import { Bot, BotDocument } from "./schema/bot.schema";

@Injectable()
export class BotService {
  private kasallikMap = new Map<number, boolean>();

  constructor(
    @InjectModel(Bot.name) private readonly botModel: Model<BotDocument>
  ) {}

  async start(ctx: Context) {
    await ctx.reply("Assalomu alaykum! Iltimos telefon raqamingizni yuboring", {
      reply_markup: {
        keyboard: [
          [{ text: "ğŸ“ Telefon raqamni yuborish", request_contact: true }],
        ],
        resize_keyboard: true,
        one_time_keyboard: true,
      },
    });
  }

  async onContact(ctx: Context) {
    const user_id = ctx.from?.id;
    if (!user_id || !ctx.message) {
      return ctx.reply("â—ï¸ Foydalanuvchi yoki xabar aniqlanmadi.");
    }

    if ("contact" in ctx.message && ctx.message.contact?.phone_number) {
      const contact = ctx.message.contact;
      let user = await this.botModel.findOne({ user_id });

      if (user) {
        user.phone_number = contact.phone_number;
        await user.save();
      } else {
        await this.botModel.create({
          user_id,
          phone_number: contact.phone_number,
          user_name: ctx.from?.username ?? "",
          first_name: ctx.from?.first_name ?? "",
          last_name: ctx.from?.last_name ?? "",
          lang: ctx.from?.language_code ?? "",
        });
      }

      await ctx.reply("âœ… Endi ismingizni yuboring", {
        reply_markup: { remove_keyboard: true },
      });
    } else {
      await ctx.reply("â—ï¸ Telefon raqami aniqlanmadi.");
    }
  }

  async onName(ctx: Context) {
    const user_id = ctx.from?.id;
    if (!user_id || !ctx.message) return;

    if ("text" in ctx.message && typeof ctx.message.text === "string") {
      const text = ctx.message.text.trim();

      if (!/^[a-zA-Z\u0400-\u04FF\s'-]{2,}$/.test(text)) {
        return ctx.reply(
          "â—ï¸ Iltimos, faqat harflardan iborat ismingizni yozing."
        );
      }

      if (/[\p{Emoji}]/u.test(text)) {
        return ctx.reply("â—ï¸ Iltimos, emoji yubormang. Faqat matn kiriting.");
      }

      const existingUser = await this.botModel.findOne({ user_id });

      if (existingUser) {
        if (!existingUser.name) {
          existingUser.name = text;
          await existingUser.save();
          await ctx.reply(`${text} ismingiz saqlandi.`);
        }
      } else {
        await this.botModel.create({
          user_id,
          name: text,
          user_name: ctx.from?.username ?? "",
          first_name: ctx.from?.first_name ?? "",
          last_name: ctx.from?.last_name ?? "",
          lang: ctx.from?.language_code ?? "",
        });

        await ctx.reply(`${text} ismingiz saqlandi.`);
      }

      await ctx.reply("ğŸ“ Endi joylashuvingizni yuboring:", {
        reply_markup: {
          keyboard: [
            [{ text: "ğŸ“ Joylashuvni yuborish", request_location: true }],
          ],
          resize_keyboard: true,
          one_time_keyboard: true,
        },
      });
    } else {
      await ctx.reply("âŒ Iltimos, faqat matn yuboring.");
    }
  }

  async handleText(ctx: Context) {
    try {
      const user_id = ctx.from?.id;
      if (!ctx.message || !("text" in ctx.message)) return;
      const text = ctx.message.text.trim();
      if (!user_id || !text) return;

      const user = await this.botModel.findOne({ user_id });

      if (!user || !user.phone_number) {
        return this.start(ctx);
      }

      // 1. Ism hali kiritilmagan boâ€˜lsa
      if (!user.name) {
        return this.onName(ctx);
      }

      // 2. âŒ Yoâ€˜q â€” sog'lom deb belgilash
      if (text === "âŒ Yoâ€˜q") {
        user.illness = "sog'lom";
        await user.save();
        await ctx.reply(
          "âœ… Farzandingiz sog'lom deb belgilandi.",
          Markup.removeKeyboard()
        );

        // Joylashuvga o'tish
        return ctx.reply("ğŸ“ Endi joylashuvingizni yuboring:", {
          reply_markup: {
            keyboard: [
              [{ text: "ğŸ“ Joylashuvni yuborish", request_location: true }],
            ],
            resize_keyboard: true,
            one_time_keyboard: true,
          },
        });
      }

      // 3. âœ… Ha â€” kasallik yozilishini boshlash
      if (text === "âœ… Ha") {
        this.kasallikMap.set(user_id, true);
        return ctx.reply(
          "ğŸ“ Farzandingizda qanday kasallik borligini yozing (masalan: gripp, allergiya)",
          Markup.removeKeyboard()
        );
      }

      // 4. Kasallik nomi faqat âœ… Ha dan keyin matn kiritilganda qabul qilinadi
      if (this.kasallikMap.has(user_id)) {
        user.illness = text;
        await user.save();
        this.kasallikMap.delete(user_id);

        await ctx.reply(`âœ… Kasallik saqlandi: ${text}`);

        return ctx.reply("ğŸ“ Endi joylashuvingizni yuboring:", {
          reply_markup: {
            keyboard: [
              [{ text: "ğŸ“ Joylashuvni yuborish", request_location: true }],
            ],
            resize_keyboard: true,
            one_time_keyboard: true,
          },
        });
      }

      // 5. Agar illness yoâ€˜q va foydalanuvchi tugma bosmasdan matn yozgan boâ€˜lsa
      if (!user.illness) {
        return ctx.reply("â—ï¸ Iltimos, quyidagi tugmalardan birini tanlang:", {
          reply_markup: {
            keyboard: [["âœ… Ha", "âŒ Yoâ€˜q"]],
            resize_keyboard: true,
            one_time_keyboard: true,
          },
        });
      }

    const user_id = ctx.from?.id;
    if (!ctx.message || !("text" in ctx.message)) return;

    const text = ctx.message.text.trim();
    console.log("ğŸ“© Kiritilgan matn:", text);

    if (!user_id || !text) return;

    const user = await this.botModel.findOne({ user_id });
    console.log("ğŸ” Foydalanuvchi:", user);

      console.log("ğŸ“ Joylashuv yoâ€˜q. onLocation() chaqirildi.");
      return this.onLocation(ctx);
    }

    if (!user.illness && !["âœ… Ha", "âŒ Yoâ€˜q"].includes(text)) {
      console.log("ğŸ©º Kasallik hali belgilanmagan. askIllnessQuestion() chaqirildi.");
      return this.askIllnessQuestion(ctx);
    }

    if (text === "âœ… Ha") {
      console.log("âœ… Ha tanlandi â€” kasallik nomi kutilmoqda.");
      this.kasallikMap.set(user_id, true);
      return ctx.reply(
        "ğŸ“ Farzandingizda qanday kasallik borligini yozing:",
        Markup.removeKeyboard()
      );
    }

    if (text === "âŒ Yoâ€˜q") {
      console.log("âŒ Yoâ€˜q tanlandi â€” sog'lom deb belgilanyapti.");
      user.illness = "sog'lom";
      await user.save();
      return ctx.reply("âœ… Farzandingiz sog'lom deb belgilandi.");
    }

    if (this.kasallikMap.has(user_id)) {
      console.log("ğŸ“ Kasallik nomi kiritilyapti:", text);
      user.illness = text;
      await user.save();
      this.kasallikMap.delete(user_id);
      await ctx.reply(`âœ… Kasallik saqlandi: ${text}`);

    // Barcha maydonlar toâ€˜ldirilganmi

      user.name &&
      user.location &&

    ) {
      console.log("ğŸ‰ Barcha bosqichlar tugadi.");
      return ctx.reply("ğŸ‰ Siz muvaffaqiyatli ro'yxatdan o'tdingiz!");
    }

    // Qolgan holatlar
    console.log("âš ï¸ Ma'lumot toâ€˜liq emas. Qayta tekshiruv.");
    if (!user.phone_number) return this.start(ctx);
    if (!user.name) return this.onName(ctx);
    if (!user.location) return this.onLocation(ctx);
    if (!user.illness) return this.askIllnessQuestion(ctx);

  } catch (error) {

    );

  async onContact(ctx: Context) {
    try {
      const user_id = ctx.from?.id;
      if (!user_id || !ctx.message) return;

      if ("contact" in ctx.message && ctx.message.contact?.phone_number) {
        const contact = ctx.message.contact;
        let user = await this.botModel.findOne({ user_id });

        if (user) {
          user.phone_number = contact.phone_number;
          await user.save();
        } else {
          user = await this.botModel.create({
            user_id,
            phone_number: contact.phone_number,
            user_name: ctx.from?.username ?? "",
            first_name: ctx.from?.first_name ?? "",
            last_name: ctx.from?.last_name ?? "",
            lang: ctx.from?.language_code ?? "",
          });
        }

        await ctx.reply("âœ… Endi ismingizni yuboring", {
          reply_markup: { remove_keyboard: true },
        });
      } else {
        await ctx.reply("â—ï¸ Telefon raqami aniqlanmadi.");
      }
    } catch (error) {
      console.error("Contact error:", error);
      await ctx.reply(
        "âš ï¸ Telefon raqamni qabul qilishda xatolik. Iltimos, qaytadan urinib ko'ring."
      );
    }
  }

  async onName(ctx: Context) {
    try {
      const user_id = ctx.from?.id;
      if (!user_id || !ctx.message || !("text" in ctx.message)) return;
      const text = ctx.message.text.trim();

      if (!/^[a-zA-Z\u0400-\u04FF\s'-]{2,}$/.test(text)) {
        return await ctx.reply(
          "â—ï¸ Iltimos, faqat harflardan iborat ismingizni yozing (kamida 2 belgi)."
        );
      }

      const user = await this.botModel.findOne({ user_id });
      if (!user) return await this.start(ctx);

      user.name = text;
      await user.save();

      await ctx.reply(`âœ… ${text} ismingiz saqlandi.`);
      await ctx.reply("ğŸ“ Endi joylashuvingizni yuboring:", {
        reply_markup: {
          keyboard: [
            [{ text: "ğŸ“ Joylashuvni yuborish", request_location: true }],
          ],
          resize_keyboard: true,
          one_time_keyboard: true,
        },
      });
    } catch (error) {
      console.error("Name error:", error);
      await ctx.reply(
        "âš ï¸ Ismni qabul qilishda xatolik. Iltimos, qaytadan urinib ko'ring."
      );
    }
  }

  async onLocation(ctx: Context) {
    try {
      const user_id = ctx.from?.id;
      if (!user_id || !ctx.message) return;

      const user = await this.botModel.findOne({ user_id });
      if (!user) return await this.start(ctx);

      if ("location" in ctx.message && ctx.message.location) {
        const { latitude, longitude } = ctx.message.location;
        user.location = `${latitude},${longitude}`;
        await user.save();

        await ctx.reply("âœ… Joylashuvingiz saqlandi.", {
          reply_markup: { remove_keyboard: true },
        });
        return await this.askIllnessQuestion(ctx);
      }

      await ctx.reply("â—ï¸ Iltimos, tugmani bosib joylashuv yuboring!", {
        reply_markup: {
          keyboard: [
            [{ text: "ğŸ“ Joylashuvni yuborish", request_location: true }],
          ],
          resize_keyboard: true,
          one_time_keyboard: true,
        },

      await ctx.reply(
        "âš ï¸ Joylashuvni qabul qilishda xatolik. Iltimos, qaytadan urinib ko'ring."
      );
    }
  }

    try {
      await ctx.reply("ğŸ©º Farzandingizda hozirda biron kasallik bormi?", {
        reply_markup: {
          keyboard: [["âœ… Ha", "âŒ Yo'q"]],

  }

  async handleText(ctx: Context) {
    try {
      const user_id = ctx.from?.id;
      if (!ctx.message || !("text" in ctx.message)) return;

      const text = ctx.message.text.trim();
      console.log("Received text:", text, "from user:", user_id);

      if (!user_id || !text) return;

      const user = await this.botModel.findOne({ user_id });
      if (!user) return await this.start(ctx);

      // Handle illness question response
      if (
        !user.illness &&
        (text === "âœ… Ha" || text === "âŒ Yo'q" || text === "âŒ YoÊ»q")
      ) {
        if (text === "âœ… Ha") {
          this.kasallikMap.set(user_id, true);
          return await ctx.reply(
            "ğŸ“ Farzandingizda qanday kasallik borligini yozing:",
            { reply_markup: { remove_keyboard: true } }
          );
        } else {
          user.illness = "sog'lom";
          await user.save();
          this.kasallikMap.delete(user_id);
          return await ctx.reply("âœ… Farzandingiz sog'lom deb belgilandi.");
        }
      }

      // Handle illness description
      if (this.kasallikMap.has(user_id)) {
        if (text.length < 2) {
          return await ctx.reply(
            "â—ï¸ Iltimos, kamida 2 ta belgidan iborat kasallik nomini yozing"
          );
        }

        user.illness = text;
        await user.save();
        this.kasallikMap.delete(user_id);
        return await ctx.reply(`âœ… Kasallik saqlandi: ${text}`);
      }

      // Check if we need to ask for missing information
      if (!user.phone_number) return await this.start(ctx);
      if (!user.name) return await this.onName(ctx);
      if (!user.location) return await this.onLocation(ctx);
      if (!user.illness) return await this.askIllnessQuestion(ctx);

      // If all information is complete
      if (user.phone_number && user.name && user.location && user.illness) {
        return await ctx.reply("ğŸ‰ Siz muvaffaqiyatli ro'yxatdan o'tdingiz!");

    } catch (error) {
